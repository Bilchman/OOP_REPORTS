using System;
using System.Threading.Channels;

namespace Lab1_var3
{
    class Program
    {
        /// <summary>
        /// Вивід інформації про автора
        /// </summary>
        static void printAuthorInfo()
        {
            Console.WriteLine("Last name: Bilchuk\nFirst name: Nikita\nAge: 18\nGroup: IPZ-14\nCourse: 1\nE-mail: bilchuk.nikita@knu.ua");
        }

        /// <summary>
        /// Обчилення значення многочлена
        /// </summary>
        static void calcPolinom()
        {
            Console.WriteLine("Calculate the value of polinom: p = 2x^5 + 3x^4 + 5x + 10");

            int a = 2;
            int b = 3;
            int c = 5;
            int d = 10;

            Console.Write("Enter x: ");
            input(out double x);

            double p = a * x * x * x * x * x + b * x * x * x * x + c * x + d;
            Console.WriteLine("Polinom is equal to " + p);
        }

        /// <summary>
        /// Обчислення значення виразу з використанням методів класу Math
        /// </summary>
        static void calcWithMath()
        {
            Console.WriteLine("Calculate the value of x = (a + b)^2 / (a - b)^2 + e^(a - 1)");

            const double e = 2.72; // константа e
            double a = 1;
            double b = 1;

            while (a == b) // перевірка на те, що a і b не рівні, щоб не було ділення на нуль
            {
                Console.Write("Enter a: ");
                input(out a);
                Console.Write("Enter b: ");
                input(out b);
                if (a == b) // якщо a і b рівні, виводимо повідомлення і просимо ввести ще раз
                    Console.WriteLine("a and b have to be different numbers, try again");
            }

            double x = Math.Pow(a + b, 2) / Math.Pow(a - b, 2) + Math.Pow(e, a - 1);
            Console.WriteLine("x is equal to " + x);
        }

        /// <summary>
        /// Обчислення значення функції, заданої формулою з умовами на x
        /// </summary>
        static void calcFunction()
        {
            Console.WriteLine("       { (sin(x))^2  , x > 0");
            Console.WriteLine("f(x) = { 1 + 2cos(x) , x < 0");
            Console.WriteLine("       {      1      , x = 0");

            Console.Write("Enter x: ");
            input(out double x);
            double f = 1; // за замовчуванням f = 1, якщо x = 0

            // обчислення значення функції в залежності від умови на x
            if (x > 0)
                f = Math.Pow(Math.Sin(x), 2);
            else if (x < 0)
                f = 1 + 2 * Math.Cos(x);

            Console.WriteLine("f(x) is equal to " + f);
        }

        /// <summary>
        /// Переведення номера місяця в пору року
        /// </summary>
        static void monthToSeason()
        {
            int m = 0;

            while (m < 1 || m > 12) // перевірка на те, що введене число є номером місяця
            {
                Console.Write("Enter month number: ");
                input(out m);
                if (m < 1 || m > 12) // якщо число не є номером місяця, виводимо повідомлення і просимо ввести ще раз
                    Console.WriteLine("It`s not a month number, try again");
            }
            
            switch (m) // визначення пори року в залежності від номера місяця
            {
                case < 3: Console.WriteLine("It`s Winter"); break;
                case < 6: Console.WriteLine("It`s Spring"); break;
                case < 9: Console.WriteLine("It`s Summer"); break;
                case < 12: Console.WriteLine("It`s Autumn"); break;
                case 12: Console.WriteLine("It`s Winter"); break;
            }
        }

        /// <summary>
        /// Обчислення суми заданого ряду з рекурентними співвідношеннями для a(k) та b(k)
        /// </summary>
        static void calcSum()
        {
            Console.WriteLine("     n                 ");
            Console.WriteLine("    ___                a(1) = 1;  a(k) = 3b(k - 1) + 1,");
            Console.WriteLine("    \\                                                             for k > 1");
            Console.WriteLine("S = /__ a(i) * b(i),   b(1) = 3;  b(k) = 1/2 * a(k) - b(k - 1),");
            Console.WriteLine("    i=1                ");

            Console.Write("Enter integer n: ");
            int n = 0;
            while (n < 1)
            {
                input(out n);
                if (n < 0)
                    Console.WriteLine("n has to be equal or more than 1.");
            }
            double a = 1;// початкове значення a при i = 1
            double b = 3;// початкове значення b при i = 1
            double sum = 3; // початкове значення суми при i = 1

            for (int i = 1; i < n; i++)// цикл для обчислення суми від i = 1 до n - 1, оскільки при i = 1 сума вже дорівнює 3
            {
                a = 3 * b + 1;
                b = 0.5 * a - b;
                sum += a * b;
            }

            Console.WriteLine("Sum is " + sum);
        }

        /// <summary>
        /// Меню програми
        /// </summary>
        static void printMenu()
        {
            Console.WriteLine("1. Author info\n2. Calculate polinom\n3. Calculate equasion with 'Math'\n4. Calculate function\n5. Convert month to season\n6. Calculate sum\n7. Exit");
        }

        /// <summary>
        /// Перевірка введення цілочисельної змінної
        /// </summary>
        /// <param name="variable"> змінна в яку вводится вихідне значення після перевірки </param>
        static void input(out int variable)
        {
            while (!int.TryParse(Console.ReadLine(), out variable))
                Console.Write("Enter integer number please: ");
        }

        /// <summary>
        /// Перевірка введення дробово-чисельної змінної
        /// </summary>
        /// <param name="variable"> змінна в яку вводится вихідне значення після перевірки </param>
        static void input(out double variable)
        {
            while (!double.TryParse(Console.ReadLine(), out variable))
                Console.Write("Enter number please: ");
        }

        /// <summary>
        /// точка входу в програму
        /// </summary>
        /// <param name="args"></param>
        static void Main(string[] args)
        {
            int choice; // змінна для збереження вибору користувача
            bool exit = false; // змінна для визначення, чи потрібно виходити з програми
            while (!exit)
            {
                printMenu();
                Console.Write("Choice: ");
                choice = Convert.ToInt32(Console.ReadLine());
                Console.WriteLine("******************************");
                switch (choice) // виконання дії в залежності від вибору користувача
                {
                    case 1: printAuthorInfo(); Console.WriteLine("******************************"); break;
                    case 2: calcPolinom(); Console.WriteLine("******************************"); break;
                    case 3: calcWithMath(); Console.WriteLine("******************************"); break;
                    case 4: calcFunction(); Console.WriteLine("******************************"); break;
                    case 5: monthToSeason(); Console.WriteLine("******************************"); break;
                    case 6: calcSum(); Console.WriteLine("******************************"); break;
                    case 7: exit = true; break;
                    default: Console.WriteLine("Invalid choice"); Console.WriteLine("******************************"); break;
                }
            }
        }
    }
}
